1.Print the following pattern using function with function prototype
*
* *
* * *
* * * *
* * * * *

Code-

#include <iostream>
using namespace std;

void pypart(int ); //function prototype

int main()
{
	int n = 5;
	pypart(n); //function call
	return 0;
}

void pypart(int n)  //function definition
{

	for (int i = 0; i < n; i++) {

		for (int j = 0; j <= i; j++) {

			cout << "* ";
		}

		cout << endl;
	}
}
